Collection pada C#

Meskipun array pada C# dapat menyimpan banyak nilai sekaligus, mereka tidak dapat mengubah ukurannya secara dinamis. Class-class collection pada C# mirip dengan Array, tetapi dapat diubah ukurannya secara dinamis. Ada banyak class yang tersedia Collections pada C#. Di bawah ini adalah beberapa contoh program yang menggunakan Collections. Akan tetapi, sebelum itu, Saya ingin memperkenalkan perulangan foreach pada C# yang sangat berguna ketika digabungkan dengan Collection. Struktur perulangan foreach adalah:

  1. foreach ([Tipe data item] [IdentifierItem] In [Collection]) {
  2.   //Hal yang dilakukan
  3.   //Contoh: Console.WriteLine([IdentifierItem]);
  4. }

List pada C#

Di bawah ini adalah contoh suatu program yang menggunakan List pada C#. Method yang penting pada List adalah Add, Remove, RemoveAt, dan Contains. Untuk method Remove dan Contains, kita perlu mengoverride method Equals.

  1. class Program {
  2.   static void Main(string[] args) {
  3.     Sales oSales = new Sales(1);
  4.     oSales.AddItem("Coca Cola", 3);
  5.     oSales.AddItem("Sprite", 3);
  6.     oSales.AddItem("Fanta", 5);
  7.     oSales.RemoveLast();
  8.     oSales.AddItem("Coca Cola", 5);
  9.     oSales.AddItem("Fanta", 3);
  10.     oSales.RemoveItem(new Sales.SalesItem() { Name = "Sprite" });
  11.     oSales.RemoveItem(new Sales.SalesItem() { Name = "Sprite" });
  12.     oSales.ShowItems();
  13.     Console.ReadKey();
  14.   }
  15. }
  16.  
  17. public class Sales {
  18.   public int ID { get; set; }
  19.   public DateTime TanggalBeli { get; set; }
  20.   public List<SalesItem> Items { get; private set; }
  21.  
  22.   public Sales(int newid) {
  23.     this.ID = newid;
  24.     TanggalBeli = DateTime.Now;
  25.     Items = new List<SalesItem>();
  26.   }
  27.  
  28.   public bool AddItem(string name, int jumlah) {
  29.     SalesItem itemBaru = new SalesItem() { Nama = name, Jumlah = jumlah };
  30.     if (!Items.Contains(itemBaru)) {
  31.       Items.Add(itemBaru);
  32.       Console.WriteLine(itemBaru.Nama " telah ditambahkan");
  33.       return true;
  34.     } else {
  35.       Console.WriteLine(itemBaru.Nama " telah ditambahkan sebelumnya");
  36.       return false;
  37.     }
  38.   }
  39.   public bool RemoveItem(SalesItem item) {
  40.     if (!Items.Contains(item)) {
  41.       Console.WriteLine("Item tidak ditemukan");
  42.       return false;
  43.     }
  44.     Items.Remove(item);
  45.     Console.WriteLine(item.Nama " telah diremove");
  46.     return true;
  47.   }
  48.   public bool RemoveLast() {
  49.     if (Items.Count == 0) {
  50.       Console.WriteLine("Tak ada item");
  51.       return false;
  52.     }
  53.     Console.WriteLine(Items[Items.Count - 1].Nama " telah diremove");
  54.     Items.RemoveAt(Items.Count - 1);
  55.     return true;  
  56.   }
  57.   public void ShowItems() {
  58.     Console.WriteLine("------- Daftar Item ----------");
  59.     foreach (SalesItem i in Items) {
  60.       Console.WriteLine(i.Nama ": " i.Jumlah);
  61.     }
  62.     Console.WriteLine("--------------------------------");
  63.   }
  64.  
  65.   public class SalesItem {
  66.     public string Nama { get; set; }
  67.     public int Jumlah { get; set; }
  68.  
  69.     //Digunakan untuk Contains dan Remove, untuk membandingkan objek
  70.     public override bool Equals(object obj) {
  71.       //Dengan var, C# akan otomatis mengetahui bahwa Other adalah SalesItem
  72.       var Banding = obj as SalesItem;
  73.       if (Banding.Nama == this.Nama) return true;
  74.       return false;
  75.     }
  76.   }
  77. }

Dictionary pada C#

Dictionary (kamus) pada C# berisi pasangan-pasangan key dan value. Tipe item pada Dictionary adalah KeyValuePair<tipeKey, tipeValue>. Berikut adalah contoh program yang menggunakan Dictionary:

  1. class Program {
  2.   static void Main(string[] args) {
  3.     Dictionary<string, string> InggrisIndonesia = new Dictionary<string, string>();
  4.  
  5.     InggrisIndonesia.Add("Get", "Ambil");
  6.     InggrisIndonesia.Add("Like", "Seperti");
  7.     InggrisIndonesia.Add("Book", "Buku");
  8.     InggrisIndonesia.Add("I", "Saya");
  9.     InggrisIndonesia.Add("Code", "Kode");
  10.     InggrisIndonesia.Add("For", "Untuk");
  11.     InggrisIndonesia.Add("Money", "Uang");
  12.  
  13.     foreach (KeyValuePair<string, string> kv in InggrisIndonesia) {
  14.       Console.WriteLine(kv.Key " = " kv.Value);
  15.     }
  16.     Console.WriteLine("---------------------------");
  17.     Console.Write("Bahasa Indonesia dari: ");
  18.     string cari = Console.ReadLine();
  19.     if (InggrisIndonesia.ContainsKey(cari)) {
  20.       Console.WriteLine(InggrisIndonesia[cari]);
  21.     } else {
  22.       Console.WriteLine("Kata tidak ditemukan");
  23.     }
  24.     Console.ReadKey();
  25.   }
  26. }

Queue pada C#

Ada 2 class Queue pada C#. Satu berada di bawah namespace System.Collections, yang lain berada di bawah namespace System.Collections.Generic. Pada program di bawah, kita menggunakan class Queue Generic:

  1. class Program {
  2.     static void Main(string[] args) {
  3.       Queue<Pelanggan> antrian = new Queue<Pelanggan>();
  4.       antrian.Enqueue(new Pelanggan("Fandi", "62852 XXXX YYY"));
  5.       antrian.Enqueue(new Pelanggan("Susanto", "62813 XXXX YYY"));
  6.       antrian.Enqueue(new Pelanggan("Billy", "62819 XXX YYYY"));
  7.  
  8.       Pelanggan Dilayani;
  9.       for (int i = 1; i <= 3; i++) {
  10.         Dilayani = antrian.Dequeue();
  11.         Console.WriteLine("Antrian No." Dilayani.NoAntrian ", Tuan." Dilayani.Nama " sedang dilayani");
  12.       }
  13.  
  14.       Console.ReadKey();
  15.     }
  16.   }
  17.  
  18.   class Pelanggan { //Access modifier tidak ditulis = internal
  19.     public static int NoAntrianBerikutnya = 1;
  20.     public string Nama { get; set; }
  21.     public string Telepon { get; set; }
  22.     public int NoAntrian { get; set; }
  23.     public Pelanggan(string nama, string telp) {
  24.       this.Nama = nama;
  25.       this.Telepon = telp;
  26.       this.NoAntrian = NoAntrianBerikutnya++;
  27.       //Berarti: NoAntrian = NoAntrianBerikutnya
  28.       //Kemudian: NoAntrianBerikutnya++
  29.     }
  30.   }
